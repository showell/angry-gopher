# Aesthetic naming alignment with Elm. Each rename brings
# a Python identifier closer to the Elm version's name (after
# camelCase/snake_case translation).
#
# Driven by ALIGNMENT_REPORT.md § 1, the `≠` rows. Picking
# the high-leverage subset where the Elm name is at least as
# good as the Python name.

# Drop the `_kind` suffix on the verb-eligibility predicates.
# Elm has bare canPeel / canPluck / canYank / canSteal / canSplitOut
# under Enumerator.elm. The Python `_kind` suffix was an artifact
# of internal naming; the args are still (kind, n, ci).
can_peel_kind -> can_peel
can_pluck_kind -> can_pluck
can_yank_kind -> can_yank
can_steal_kind -> can_steal
can_split_out_kind -> can_split_out

# `partial_ok` is a Boolean predicate; Elm names it `isPartialOk`.
partial_ok -> is_partial_ok

# `describe_move(desc)` is more naturally `move.describe(desc)`
# in Python — the module-qualifier handles the "_move" suffix.
# Mirrors Elm's `Move.describe`.
describe_move -> describe

# `without` is a generic name that doesn't say WHAT we drop.
# `without_at` mirrors Elm's `withoutAt` and is self-documenting.
without -> without_at

# `label_d` is awkward — the `_d` was for "with deck suffix"
# but Elm just calls it `cardLabel` and always emits the
# deck if present. Rename for consistency.
label_d -> card_label
